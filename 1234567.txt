ВЫВЕСТИ ВСЕХ КЛИЕНТОВ КОТОРЫЕ НЕ КУПИЛИ НИ ОДНОЙ ПУТЁВКИ
select Клиент.*
from клиент
where not exists
(select ТурПутевка.*
from ТурПутевка
where ТурПутевка.id_клиент=Клиент.id)

НЕ КОРРЕЛИРОВАННЫЙ ПОДЗАПРОС

select r.фамилия
from работник r
where r.id in
(select tp.id_работник 
from ТурПутевка tp
where tp.id_отель in
(select o.id
from Отель o
where o.название like '%PHUKET%'))

КОРРЕЛИРОВАННЫЙ с exists

select r.фамилия
from работник r
where EXISTS
(select tp.*
from ТурПутевка tp
where r.id=tp.id_работник and
tp.id_отель in
(select o.id
from Отель o
where o.название like '%PHUKET%'))
